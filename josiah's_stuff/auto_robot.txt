#!/usr/bin/env python3
"""
AUTO4508 - Autonomous Robot Node
Pioneer 3-AT with Lakibeam LiDAR (192.168.198.2)

Sequence:
  1. Wait STARTUP_DELAY_S seconds
  2. Run startup motion sequence (forward, back, turns)
  3. Wait for LiDAR confirmation
  4. Drive forward autonomously
  5. Stop if obstacle detected within OBSTACLE_DISTANCE_M in front
  6. Resume when path is clear
"""

import time
import math
import threading

import rclpy
from rclpy.node import Node
from rclpy.executors import MultiThreadedExecutor
from rclpy.callback_groups import ReentrantCallbackGroup

from geometry_msgs.msg import Twist
from sensor_msgs.msg import LaserScan
from std_msgs.msg import String


# ── Tuning constants ──────────────────────────────────────────────────────────
OBSTACLE_DISTANCE_M  = 1.0    # Stop if anything closer than this (metres)
CHASSIS_MIN_RANGE_M  = 0.5    # Ignore readings closer than this (robot body)
FORWARD_SPEED        = 0.2    # m/s for autonomous forward drive
FORWARD_CONE_DEG     = 90     # Half-angle of danger zone — covers full front hemisphere
STARTUP_DELAY_S      = 3.0    # Seconds to wait before starting sequence
# ─────────────────────────────────────────────────────────────────────────────


class AutoRobot(Node):

    def __init__(self):
        super().__init__("auto_robot")

        # ReentrantCallbackGroup allows LiDAR callback to fire while timer
        # callback is blocked inside _run_motion()
        self.cb_group = ReentrantCallbackGroup()

        # ── Publishers ────────────────────────────────────────────────────────
        self.cmd_pub    = self.create_publisher(Twist,  "/cmd_vel",      10)
        self.status_pub = self.create_publisher(String, "/robot_status", 10)

        # ── LiDAR subscriber ─────────────────────────────────────────────────
        self.lidar_sub = self.create_subscription(
            LaserScan,
            "/scan",
            self._lidar_callback,
            10,
            callback_group=self.cb_group,
        )

        # ── State (protected by lock — two threads access these) ──────────────
        self._lock          = threading.Lock()
        self.lidar_ready    = False
        self.obstacle_ahead = False
        self.phase          = "WAITING"
        self.start_time     = time.time()

        # ── Main loop timer (10 Hz) ───────────────────────────────────────────
        self.timer = self.create_timer(
            0.1, self._loop, callback_group=self.cb_group
        )

        self._log("Auto robot node started — waiting %.1f s before sequence…"
                  % STARTUP_DELAY_S)

    # ── Helpers ───────────────────────────────────────────────────────────────

    def _log(self, msg: str):
        self.get_logger().info(msg)
        s = String()
        s.data = "[%s] %s" % (self.phase, msg)
        self.status_pub.publish(s)

    def _send_cmd(self, lin: float = 0.0, ang: float = 0.0):
        msg = Twist()
        msg.linear.x  = lin
        msg.angular.z = ang
        self.cmd_pub.publish(msg)

    def _stop(self):
        try:
            self._send_cmd(0.0, 0.0)
        except Exception:
            pass  # Suppress errors during shutdown

    # ── LiDAR callback ────────────────────────────────────────────────────────

    def _lidar_callback(self, msg: LaserScan):
        # Log first scan confirmation
        with self._lock:
            if not self.lidar_ready:
                self.lidar_ready = True
                self._log(
                    "LiDAR is ON — first scan received (%.0f rays, "
                    "%.1f deg to %.1f deg)"
                    % (
                        len(msg.ranges),
                        math.degrees(msg.angle_min),
                        math.degrees(msg.angle_max),
                    )
                )

        # Check forward hemisphere (±FORWARD_CONE_DEG) for obstacles
        angle_min = msg.angle_min
        angle_inc = msg.angle_increment
        cone_rad  = math.radians(FORWARD_CONE_DEG)

        obstacle_found = False
        for i, r in enumerate(msg.ranges):
            angle = angle_min + i * angle_inc

            # Only look at rays in the forward cone
            if abs(angle) > cone_rad:
                continue

            # Ignore invalid readings (inf, nan)
            if math.isnan(r) or math.isinf(r):
                continue

            # Ignore out-of-sensor-range readings
            # Note: only check range_max since range_min is 0.0 for Lakibeam
            if r > msg.range_max:
                continue

            # Ignore the robot's own chassis and mounting hardware
            if r < CHASSIS_MIN_RANGE_M:
                continue

            # Obstacle detected within threshold
            if r < OBSTACLE_DISTANCE_M:
                obstacle_found = True
                break

        with self._lock:
            prev = self.obstacle_ahead
            self.obstacle_ahead = obstacle_found

        if obstacle_found and not prev:
            self._log("OBSTACLE within %.1f m — stopping!" % OBSTACLE_DISTANCE_M)
            self._stop()
        elif not obstacle_found and prev:
            with self._lock:
                phase = self.phase
            if phase == "DRIVING":
                self._log("Path clear — resuming forward drive")

    # ── Blocking motion helper ────────────────────────────────────────────────

    def _run_motion(self, lin: float, ang: float, duration: float, label: str):
        """
        Sends velocity commands for `duration` seconds.
        Using time.sleep() is safe here because MultiThreadedExecutor
        runs the LiDAR callback in a separate thread.
        """
        self._log(label)
        start = time.time()
        while time.time() - start < duration:
            self._send_cmd(lin, ang)
            time.sleep(0.05)
        self._stop()
        time.sleep(0.3)

    # ── Startup motion sequence ───────────────────────────────────────────────

    def _run_sequence(self):
        self._log("=== Starting startup motion sequence ===")

        self._run_motion( 0.2,  0.0, 1.5, "Forward")
        self._run_motion(-0.2,  0.0, 1.5, "Backward")
        self._run_motion( 0.0, -0.4, 1.0, "Turn right (approx 45 deg)")
        self._run_motion( 0.0,  0.4, 2.0, "Turn left (approx 135 deg)")
        self._run_motion( 0.0, -0.4, 1.0, "Return to centre (approx 90 deg)")

        self._log("=== Startup sequence complete ===")

    # ── Wait for LiDAR ───────────────────────────────────────────────────────

    def _wait_for_lidar(self):
        self._log("Waiting for LiDAR scan...")
        timeout = time.time() + 10.0
        while True:
            with self._lock:
                ready = self.lidar_ready
            if ready:
                break
            if time.time() > timeout:
                self._log("WARNING: LiDAR timeout — continuing without confirmation")
                break
            time.sleep(0.1)

    # ── Main loop (timer callback) ────────────────────────────────────────────

    def _loop(self):
        with self._lock:
            phase    = self.phase
            obstacle = self.obstacle_ahead

        if phase == "WAITING":
            if time.time() - self.start_time >= STARTUP_DELAY_S:
                with self._lock:
                    self.phase = "SEQUENCE"
                self._run_sequence()
                self._wait_for_lidar()
                with self._lock:
                    self.phase = "DRIVING"
                self._log(
                    "Autonomous forward drive started "
                    "(obstacle threshold = %.1f m, "
                    "forward cone = %.0f deg)"
                    % (OBSTACLE_DISTANCE_M, FORWARD_CONE_DEG)
                )

        elif phase == "SEQUENCE":
            pass

        elif phase == "DRIVING":
            if obstacle:
                with self._lock:
                    self.phase = "STOPPED"
                self._stop()
            else:
                self._send_cmd(lin=FORWARD_SPEED)

        elif phase == "STOPPED":
            self._stop()
            if not obstacle:
                with self._lock:
                    self.phase = "DRIVING"
                self._log("Path clear — resuming autonomous drive")


# ── Entry point ───────────────────────────────────────────────────────────────

def main(args=None):
    rclpy.init(args=args)
    node = AutoRobot()

    # MultiThreadedExecutor: 2 threads — one for timer, one for LiDAR callback
    executor = MultiThreadedExecutor(num_threads=2)
    executor.add_node(node)

    try:
        executor.spin()
    except KeyboardInterrupt:
        pass
    finally:
        node._stop()
        node.destroy_node()
        try:
            rclpy.shutdown()
        except Exception:
            pass


if __name__ == "__main__":
    main()
