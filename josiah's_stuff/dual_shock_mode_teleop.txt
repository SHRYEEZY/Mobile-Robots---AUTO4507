#!/usr/bin/env python3
"""
DualShock teleop + autonomous mode with startup sequence for Pioneer.
Modes:
  - MANUAL: joystick drives robot, only while manual deadman is held
  - AUTO: runs autonomous startup sequence then forward drive with obstacle avoidance
"""
import math
import time
import threading
from typing import Optional

import rclpy
from rclpy.node import Node
from rclpy.executors import MultiThreadedExecutor
from rclpy.callback_groups import ReentrantCallbackGroup

from geometry_msgs.msg import Twist
from sensor_msgs.msg import Joy, LaserScan
from std_msgs.msg import String


# ── Auto mode tuning constants ────────────────────────────────────────────────
OBSTACLE_DISTANCE_M  = 1.0    # Stop if anything closer than this (metres)
CHASSIS_MIN_RANGE_M  = 0.5    # Ignore readings closer than this (robot body)
FORWARD_SPEED        = 0.2    # m/s for autonomous forward drive
FORWARD_CONE_DEG     = 90     # Half-angle of danger zone — covers full front hemisphere
STARTUP_DELAY_S      = 3.0    # Seconds to wait before starting sequence
# ──────────────────────────────────────────────────────────────────────────────


class DualShockModeTeleop(Node):
    MODE_MANUAL = "manual"
    MODE_AUTO = "auto"
    
    def __init__(self) -> None:
        super().__init__("dualshock_mode_teleop")
        
        # ReentrantCallbackGroup allows LiDAR callback to fire while timer
        # callback is blocked inside motion sequences
        self.cb_group = ReentrantCallbackGroup()
        
        # ---------------- Parameters ----------------
        self.declare_parameter("cmd_vel_topic", "/cmd_vel")
        self.declare_parameter("joy_topic", "/joy")
        self.declare_parameter("mode_topic", "/control_mode")
        self.declare_parameter("status_topic", "/robot_status")
        self.declare_parameter("scan_topic", "/scan")
        self.declare_parameter("max_linear", 0.50)
        self.declare_parameter("max_angular", 2.00)
        self.declare_parameter("deadzone", 0.10)
        self.declare_parameter("publish_rate_hz", 20.0)
        
        # Button/axis mapping
        self.declare_parameter("btn_x", 0)              # X
        self.declare_parameter("btn_circle", 1)         # O
        self.declare_parameter("axis_left_x", 0)        # left stick left/right
        self.declare_parameter("axis_right_y", 3)       # right stick up/down
        self.declare_parameter("axis_l2", 4)            # L2 trigger axis
        self.declare_parameter("axis_r2", 5)            # R2 trigger axis
        
        self.cmd_vel_topic = self.get_parameter("cmd_vel_topic").value
        self.joy_topic = self.get_parameter("joy_topic").value
        self.mode_topic = self.get_parameter("mode_topic").value
        self.status_topic = self.get_parameter("status_topic").value
        self.scan_topic = self.get_parameter("scan_topic").value
        self.max_linear = float(self.get_parameter("max_linear").value)
        self.max_angular = float(self.get_parameter("max_angular").value)
        self.deadzone = float(self.get_parameter("deadzone").value)
        self.publish_rate_hz = float(self.get_parameter("publish_rate_hz").value)
        
        self.btn_x = int(self.get_parameter("btn_x").value)
        self.btn_circle = int(self.get_parameter("btn_circle").value)
        self.axis_left_x = int(self.get_parameter("axis_left_x").value)
        self.axis_right_y = int(self.get_parameter("axis_right_y").value)
        self.axis_l2 = int(self.get_parameter("axis_l2").value)
        self.axis_r2 = int(self.get_parameter("axis_r2").value)
        
        # ---------------- State (protected by lock) ----------------
        self._lock = threading.Lock()
        self.mode = self.MODE_MANUAL
        self.last_joy: Optional[Joy] = None
        
        # Auto mode state
        self.lidar_ready = False
        self.obstacle_ahead = False
        self.auto_phase = "WAITING"  # WAITING, SEQUENCE, DRIVING, STOPPED
        self.auto_start_time = None
        self.auto_mode_active = False  # Track if we're running auto mode
        
        # ---------------- ROS interfaces ----------------
        self.cmd_pub = self.create_publisher(Twist, self.cmd_vel_topic, 10)
        self.mode_pub = self.create_publisher(String, self.mode_topic, 10)
        self.status_pub = self.create_publisher(String, self.status_topic, 10)
        
        self.joy_sub = self.create_subscription(
            Joy,
            self.joy_topic,
            self.joy_callback,
            10,
            callback_group=self.cb_group
        )
        
        self.lidar_sub = self.create_subscription(
            LaserScan,
            self.scan_topic,
            self._lidar_callback,
            10,
            callback_group=self.cb_group,
        )
        
        timer_period = 1.0 / self.publish_rate_hz
        self.timer = self.create_timer(
            timer_period, self.control_loop, callback_group=self.cb_group
        )
        
        self.get_logger().info("DualShock teleop node with auto mode started.")
        self.get_logger().info("Default mode: MANUAL")
    
    # ── Helpers ───────────────────────────────────────────────────────────────
    
    def _log(self, msg: str):
        """Log message and publish to status topic"""
        with self._lock:
            phase = self.auto_phase if self.mode == self.MODE_AUTO else self.mode
        self.get_logger().info(msg)
        s = String()
        s.data = f"[{phase}] {msg}"
        self.status_pub.publish(s)
    
    def apply_deadzone(self, value: float) -> float:
        if abs(value) < self.deadzone:
            return 0.0
        return value
    
    def axis_pressed(self, joy: Joy, axis_index: int, threshold: float = 0.5) -> bool:
        """Trigger handling for deadman switches"""
        if axis_index >= len(joy.axes):
            return False
        return joy.axes[axis_index] < threshold
    
    def button_pressed(self, joy: Joy, button_index: int) -> bool:
        if button_index >= len(joy.buttons):
            return False
        return joy.buttons[button_index] == 1
    
    def zero_twist(self) -> Twist:
        msg = Twist()
        msg.linear.x = 0.0
        msg.angular.z = 0.0
        return msg
    
    def _send_cmd(self, lin: float = 0.0, ang: float = 0.0):
        msg = Twist()
        msg.linear.x = lin
        msg.angular.z = ang
        self.cmd_pub.publish(msg)
    
    def _stop(self):
        try:
            self._send_cmd(0.0, 0.0)
        except Exception:
            pass  # Suppress errors during shutdown
    
    # ── LiDAR callback ────────────────────────────────────────────────────────
    
    def _lidar_callback(self, msg: LaserScan):
        # Log first scan confirmation
        with self._lock:
            if not self.lidar_ready:
                self.lidar_ready = True
                self._log(
                    f"LiDAR is ON — first scan received ({len(msg.ranges)} rays, "
                    f"{math.degrees(msg.angle_min):.1f} deg to {math.degrees(msg.angle_max):.1f} deg)"
                )
        
        # Check forward hemisphere for obstacles
        angle_min = msg.angle_min
        angle_inc = msg.angle_increment
        cone_rad = math.radians(FORWARD_CONE_DEG)
        
        obstacle_found = False
        for i, r in enumerate(msg.ranges):
            angle = angle_min + i * angle_inc
            
            # Only look at rays in the forward cone
            if abs(angle) > cone_rad:
                continue
            
            # Ignore invalid readings
            if math.isnan(r) or math.isinf(r):
                continue
            
            # Ignore out-of-sensor-range readings
            if r > msg.range_max:
                continue
            
            # Ignore the robot's own chassis
            if r < CHASSIS_MIN_RANGE_M:
                continue
            
            # Obstacle detected within threshold
            if r < OBSTACLE_DISTANCE_M:
                obstacle_found = True
                break
        
        with self._lock:
            prev = self.obstacle_ahead
            self.obstacle_ahead = obstacle_found
            mode = self.mode
            phase = self.auto_phase
        
        # Only react to obstacles in AUTO mode
        if mode == self.MODE_AUTO:
            if obstacle_found and not prev:
                self._log(f"OBSTACLE within {OBSTACLE_DISTANCE_M:.1f} m — stopping!")
                self._stop()
            elif not obstacle_found and prev:
                if phase == "DRIVING":
                    self._log("Path clear — resuming forward drive")
    
    # ── Blocking motion helper for auto sequences ─────────────────────────────
    
    def _run_motion(self, lin: float, ang: float, duration: float, label: str):
        """Sends velocity commands for duration seconds"""
        self._log(label)
        start = time.time()
        while time.time() - start < duration:
            # Check if manual mode was activated during sequence
            with self._lock:
                if self.mode == self.MODE_MANUAL:
                    self._log("Manual mode activated - aborting sequence")
                    self._stop()
                    return
            self._send_cmd(lin, ang)
            time.sleep(0.05)
        self._stop()
        time.sleep(0.3)
    
    # ── Startup motion sequence ───────────────────────────────────────────────
    
    def _run_sequence(self):
        self._log("=== Starting startup motion sequence ===")
        
        self._run_motion(0.2, 0.0, 1.5, "Forward")
        self._run_motion(-0.2, 0.0, 1.5, "Backward")
        self._run_motion(0.0, -0.4, 1.0, "Turn right (approx 45 deg)")
        self._run_motion(0.0, 0.4, 2.0, "Turn left (approx 135 deg)")
        self._run_motion(0.0, -0.4, 1.0, "Return to centre (approx 90 deg)")
        
        self._log("=== Startup sequence complete ===")
    
    # ── Wait for LiDAR ───────────────────────────────────────────────────────
    
    def _wait_for_lidar(self):
        self._log("Waiting for LiDAR scan...")
        timeout = time.time() + 10.0
        while True:
            with self._lock:
                ready = self.lidar_ready
                if self.mode == self.MODE_MANUAL:
                    self._log("Manual mode activated - aborting LiDAR wait")
                    return False
            if ready:
                return True
            if time.time() > timeout:
                self._log("WARNING: LiDAR timeout — continuing without confirmation")
                return True
            time.sleep(0.1)
    
    # ── Callbacks ────────────────────────────────────────────────────────────
    
    def joy_callback(self, msg: Joy) -> None:
        self.last_joy = msg
        
        # Mode switching
        if self.button_pressed(msg, self.btn_circle):
            with self._lock:
                if self.mode != self.MODE_MANUAL:
                    self.mode = self.MODE_MANUAL
                    self.auto_mode_active = False
                    self.get_logger().info("Switched to MANUAL mode")
                    self.publish_mode()
                    self._stop()
        
        if self.button_pressed(msg, self.btn_x):
            with self._lock:
                if self.mode != self.MODE_AUTO:
                    self.mode = self.MODE_AUTO
                    self.auto_phase = "WAITING"
                    self.auto_start_time = time.time()
                    self.auto_mode_active = True
                    self.get_logger().info("Switched to AUTO mode")
                    self.publish_mode()
    
    def publish_mode(self) -> None:
        mode_msg = String()
        mode_msg.data = self.mode
        self.mode_pub.publish(mode_msg)
    
    # ── Main control loop (timer callback) ────────────────────────────────────
    
    def control_loop(self) -> None:
        if self.last_joy is None:
            self.cmd_pub.publish(self.zero_twist())
            return
        
        joy = self.last_joy
        
        with self._lock:
            mode = self.mode
            obstacle = self.obstacle_ahead
            phase = self.auto_phase
        
        # ── MANUAL MODE ──────────────────────────────────────────────────────
        if mode == self.MODE_MANUAL:
            manual_deadman = self.axis_pressed(joy, self.axis_l2)
            if not manual_deadman:
                self.cmd_pub.publish(self.zero_twist())
                return
            
            linear_input = 0.0
            angular_input = 0.0
            
            if self.axis_right_y < len(joy.axes):
                linear_input = joy.axes[self.axis_right_y]
            
            if self.axis_left_x < len(joy.axes):
                angular_input = joy.axes[self.axis_left_x]
            
            linear_input = self.apply_deadzone(linear_input)
            angular_input = self.apply_deadzone(angular_input)
            
            cmd = Twist()
            cmd.linear.x = self.max_linear * linear_input
            cmd.angular.z = self.max_angular * angular_input
            self.cmd_pub.publish(cmd)
            return
        
        # ── AUTO MODE ────────────────────────────────────────────────────────
        if mode == self.MODE_AUTO:
            auto_deadman = self.axis_pressed(joy, self.axis_r2)
            if not auto_deadman:
                self.cmd_pub.publish(self.zero_twist())
                return
            
            # State machine for autonomous behavior
            if phase == "WAITING":
                with self._lock:
                    if self.auto_start_time and time.time() - self.auto_start_time >= STARTUP_DELAY_S:
                        self.auto_phase = "SEQUENCE"
                        # Run sequence in this thread (timer uses ReentrantCallbackGroup)
                        self._run_sequence()
                        
                        # Wait for LiDAR
                        if self._wait_for_lidar():
                            with self._lock:
                                self.auto_phase = "DRIVING"
                            self._log(
                                f"Autonomous forward drive started "
                                f"(obstacle threshold = {OBSTACLE_DISTANCE_M:.1f} m, "
                                f"forward cone = {FORWARD_CONE_DEG:.0f} deg)"
                            )
                        else:
                            # Manual mode was activated during wait
                            return
            
            elif phase == "SEQUENCE":
                # Currently running sequence, do nothing
                pass
            
            elif phase == "DRIVING":
                if obstacle:
                    with self._lock:
                        self.auto_phase = "STOPPED"
                    self._stop()
                else:
                    self._send_cmd(lin=FORWARD_SPEED)
            
            elif phase == "STOPPED":
                self._stop()
                if not obstacle:
                    with self._lock:
                        self.auto_phase = "DRIVING"
                    self._log("Path clear — resuming autonomous drive")


def main(args=None) -> None:
    rclpy.init(args=args)
    node = DualShockModeTeleop()
    
    # MultiThreadedExecutor: allows LiDAR and joy callbacks to run concurrently
    executor = MultiThreadedExecutor(num_threads=3)
    executor.add_node(node)
    
    try:
        executor.spin()
    except KeyboardInterrupt:
        pass
    finally:
        node._stop()
        node.destroy_node()
        try:
            rclpy.shutdown()
        except Exception:
            pass


if __name__ == "__main__":
    main()
